\begin{adjustwidth}{20pt}{}
    Frequenzgang:
\end{adjustwidth}
\begin{align*}

    &= \frac{\frac{1}{\omega C}}{\frac{1}{\omega C} - \omega L}\\
    &= \frac{1}{1 - \omega^2 L C}\\
