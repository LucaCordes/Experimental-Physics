    &= - B \deriv{A}{t}\\

    \\
    0 &= \Delta W_{pot} + \Delta W_{kin} + \Delta W_{el}\\
    0 &= m g x + \frac m2 v^2 + \int I_{ind} U_{ind} \dt\\

    0 &= m g v + m v a + \frac{B^2 d^2 }{R} v^2 \\
    a &= - g - \frac{B^2 d^2 }{m R} v \\
    F_{ges} &= \underbrace{- m g}_{F_{g}} + \underbrace{ \frac{-B^2 d^2 }{R} v}_{F_{el}} \\
    \\
